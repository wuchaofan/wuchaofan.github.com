def partition(arr,left,right):
	i = left + 1
	j = right
	first = arr[left]
	while True:

		while i<=j and arr[i]<=first: i+=1
		while j>=i and arr[j]>=first: j-=1

		if i>j:
			break

		arr[i],arr[j] = arr[j],arr[i]
	arr[left],arr[j] = arr[j],arr[left]

	return j


def _qsort(arr,left,right):
	if left < right:

		point = partition(arr,left,right)

		_qsort(arr,left,point-1)
		_qsort(arr,point+1,right)

def Qsort(arr):
	_qsort(arr,0,len(arr)-1)
	return arr
